// user.h
#pragma once
#include <sqlite_orm/sqlite_orm.h>
#include <string>

struct User {
    int user_id;
    std::string firstName;
    std::string lastName;
    std::string phone;
    std::string email;
};

inline auto create_database(const std::string& path) {
    using namespace sqlite_orm;
    return make_storage(path,
        make_table("users",
            make_column("user_id", &User::user_id, primary_key()),
            make_column("firstName", &User::firstName),
            make_column("lastName", &User::lastName),
            make_column("phone", &User::phone),
            make_column("email", &User::email)
        )
    );
}Copied!
